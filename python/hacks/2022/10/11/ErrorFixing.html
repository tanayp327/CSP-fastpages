menu =  {"burger": 3.99,
         "fries": 2.99,
         "drink": 0.99}
total = 0

print("Menu:")
for k,v in menu.items():
    print(k + "  $" + str(v)) 

order = True
while order:   
    item = input("choose your order")
    if item in menu.keys(): 
        total+= menu[item]
        print("Your order so far: 1", item)
        print("This costs $", menu[item])
    elif item == str("Done"):
        total = round(total, 3)
        print("Your total is: $", total)
        print("Enjoy!")
    else:           
        order = False
    
        
print("Have a good day!")
